// Prepare data for Daily Views chart (ensuring all days are present)
const formatChartData = (dailyViewsData = [], timeframeDays = 7) => {
  const dataMap = new Map(dailyViewsData.map((item) => [item.date, item]));
  const result = [];
  const today = new Date();

  for (let i = 0; i < timeframeDays; i++) {
    const date = new Date(today);
    date.setDate(today.getDate() - i);
    date.setHours(0, 0, 0, 0); // Normalize date
    const dateString = date.toISOString().split("T")[0];

    const dayData = dataMap.get(dateString) || { count: 0, uniqueCount: 0 };
    result.push({
      date: dateString,
      count: dayData.count || 0,
      uniqueCount: dayData.uniqueCount || 0,
    });
  }
  return result.reverse(); // Sort from oldest to newest
};

// Calculate period-over-period change percentage
const calculateChange = (current, previous) => {
  if (previous === null || previous === undefined || previous === 0) {
    return current > 0 ? Infinity : 0; // Use Infinity to indicate growth from zero
  }
  if (current === null || current === undefined) return 0;
  return ((current - previous) / previous) * 100;
};

// Calculate metrics based on fetched stats and timeframe
const calculatePeriodMetrics = (dailyViewsData = [], timeframeDays = 7) => {
  const now = new Date();
  now.setHours(0, 0, 0, 0);
  const currentPeriodStart = new Date(now);
  currentPeriodStart.setDate(now.getDate() - timeframeDays + 1);
  const previousPeriodStart = new Date(currentPeriodStart);
  previousPeriodStart.setDate(currentPeriodStart.getDate() - timeframeDays);

  let currentViews = 0,
    previousViews = 0,
    currentUnique = 0,
    previousUnique = 0;

  dailyViewsData.forEach((item) => {
    const itemDate = new Date(item.date);
    itemDate.setHours(0, 0, 0, 0); // Normalize date

    if (itemDate >= currentPeriodStart && itemDate <= now) {
      currentViews += item.count || 0;
      currentUnique += item.uniqueCount || 0;
    } else if (
      itemDate >= previousPeriodStart &&
      itemDate < currentPeriodStart
    ) {
      previousViews += item.count || 0;
      previousUnique += item.uniqueCount || 0;
    }
  });

  return {
    currentViews,
    previousViews,
    viewsChange: calculateChange(currentViews, previousViews),
    currentUnique,
    previousUnique,
    uniqueChange: calculateChange(currentUnique, previousUnique),
  };
};

const BaseTooltip = ({ children }) => (

const DailyViewsTooltip = ({ active, payload, label }) => {
  if (active && payload && payload.length && label) {
    const formattedDate = new Date(label).toLocaleDateString("en-US", {
      weekday: "short",
      month: "short",
      day: "numeric",
    });
    return (
      <BaseTooltip>
        <p className="font-medium text-indigo-200 mb-2">{formattedDate}</p>
        {payload.map((entry, index) => (
          <div key={index} className="flex items-center justify-between mb-1">
            <div className="flex items-center">
              <div
                className="w-2.5 h-2.5 rounded-full mr-2"
                style={{ backgroundColor: entry.color || entry.stroke }}
              />
              <span className="text-gray-300">{entry.name}:</span>
            </div>
            <span
              className="font-medium ml-3"
              style={{ color: entry.color || entry.stroke }}
            >
              {entry.value?.toLocaleString()}
            </span>
          </div>
        ))}
      </BaseTooltip>
    );
  }
  return null;
};

const SimplePieTooltip = ({ active, payload }) => {
  if (active && payload && payload.length) {
    const data = payload[0].payload;
    const name = data.device || data.country || data.source || "Unknown";
    const percentage = ((data.count / data.total) * 100).toFixed(1);
    return (
      <BaseTooltip>
        <p className="font-medium text-indigo-200 mb-1 capitalize">{name}</p>
        <p className="text-gray-300">
          Views:{" "}
          <span className="font-medium text-white">
            {data.count?.toLocaleString()}
          </span>{" "}
          ({percentage}%)
        </p>
      </BaseTooltip>
    );
  }
  return null;
};

                          fontSize: "11px",
                        }}
                        formatter={(value) => (
                          <span style={{ color: "#D1D5DB" }}>{value}</span>
                        )}
                      />
                      <defs>
                        <linearGradient
                          id="gradCount"
                          x1="0"
                          y1="0"
                          x2="0"
                          y2="1"
                        >
                          <stop
                            offset="0%"
                            stopColor="#818cf8"
                            stopOpacity={0.8}
                          />
                          <stop
                            offset="100%"
                            stopColor="#818cf8"
                            stopOpacity={0.3}
                          />
                        </linearGradient>
                        <linearGradient
                          id="gradUnique"
                          x1="0"
                          y1="0"
                          x2="0"
                          y2="1"
                        >
                          <stop
                            offset="0%"
                            stopColor="#34d399"
                            stopOpacity={0.8}
                          />
                          <stop
                            offset="100%"
                            stopColor="#34d399"
                            stopOpacity={0.3}
                          />
                        </linearGradient>
                      </defs>
                      <Bar
                        dataKey="count"
                        name="Total"
                        fill="url(#gradCount)"
                        radius={[4, 4, 0, 0]}
                        barSize={10}
                        animationDuration={CHART_ANIMATION_DURATION}
                      />
                      <Bar
                        dataKey="uniqueCount"
                        name="Unique"
                        fill="url(#gradUnique)"
                        radius={[4, 4, 0, 0]}
                        barSize={10}
                        animationDuration={CHART_ANIMATION_DURATION}
                        animationBegin={200}
                      />
                    </BarChart>
                  )}
                  {activeChart === "area" && (
                    <AreaChart
                      data={formattedChartData}
                      margin={{ top: 5, right: 5, left: -25, bottom: 0 }}
                    >
                      <CartesianGrid
                        strokeDasharray="3 3"
                        stroke="#4B5563"
                        strokeOpacity={0.3}
                        vertical={false}
                      />
                      <XAxis
                        dataKey="date"
                        axisLine={false}
                        tickLine={false}
                        tick={{ fill: "#9ca3af", fontSize: 10 }}
                        dy={10}
                        tickFormatter={formatDateTick}
                      />
                      <YAxis
                        axisLine={false}
                        tickLine={false}
                        tick={{ fill: "#9ca3af", fontSize: 10 }}
                        width={30}
                      />
                      <Tooltip
                        content={<DailyViewsTooltip />}
                        cursor={{
                          stroke: "rgba(156, 163, 175, 0.3)",
                          strokeWidth: 1,
                        }}
                      />
                      <Legend
                        wrapperStyle={{
                          color: "#d1d5db",
                          paddingTop: "10px",
                          fontSize: "11px",
                        }}
                        formatter={(value) => (
                          <span style={{ color: "#D1D5DB" }}>{value}</span>
                        )}
                      />
                      <defs>
                        <linearGradient
                          id="areaCount"
                          x1="0"
                          y1="0"
                          x2="0"
                          y2="1"
                        >
                          <stop
                            offset="5%"
                            stopColor="#818cf8"
                            stopOpacity={0.4}
                          />
                          <stop
                            offset="95%"
                            stopColor="#818cf8"
                            stopOpacity={0}
                          />
                        </linearGradient>
                        <linearGradient
                          id="areaUnique"
                          x1="0"
                          y1="0"
                          x2="0"
                          y2="1"
                        >
                          <stop
                            offset="5%"
                            stopColor="#34d399"
                            stopOpacity={0.4}
                          />
                          <stop
                            offset="95%"
                            stopColor="#34d399"
                            stopOpacity={0}
                          />
                        </linearGradient>
                      </defs>
                      <Area
                        type="monotone"
                        dataKey="count"
                        name="Total"
                        stroke="#818cf8"
                        strokeWidth={2}
                        fill="url(#areaCount)"
                        activeDot={{ r: 5, strokeWidth: 1, stroke: "#fff" }}
                        animationDuration={CHART_ANIMATION_DURATION}
                      />
                      <Area
                        type="monotone"
                        dataKey="uniqueCount"
                        name="Unique"
                        stroke="#34d399"
                        strokeWidth={2}
                        fill="url(#areaUnique)"
                        activeDot={{ r: 5, strokeWidth: 1, stroke: "#fff" }}
                        animationDuration={CHART_ANIMATION_DURATION}
                        animationBegin={200}
                      />
                    </AreaChart>
                  )}
                  {activeChart === "combined" && (
                    <ComposedChart
                      data={formattedChartData}
                      margin={{ top: 5, right: 5, left: -25, bottom: 0 }}
                    >
                      <CartesianGrid
                        strokeDasharray="3 3"
                        stroke="#4B5563"
                        strokeOpacity={0.3}
                        vertical={false}
                      />
                      <XAxis
                        dataKey="date"
                        axisLine={false}
                        tickLine={false}
                        tick={{ fill: "#9ca3af", fontSize: 10 }}
                        dy={10}
                        tickFormatter={formatDateTick}
                      />
                      <YAxis
                        yAxisId="left"
                        axisLine={false}
                        tickLine={false}
                        tick={{ fill: "#9ca3af", fontSize: 10 }}
                        width={30}
                      />
                      <YAxis
                        yAxisId="right"
                        orientation="right"
                        axisLine={false}
                        tickLine={false}
                        tick={{ fill: "#a78bfa", fontSize: 10 }}
                        width={30}
                      />
                      <Tooltip
                        content={<DailyViewsTooltip />}
                        cursor={{ fill: "rgba(79, 70, 229, 0.1)" }}
                      />
                      <Legend
                        wrapperStyle={{
                          color: "#d1d5db",
                          paddingTop: "10px",
                          fontSize: "11px",
                        }}
                        formatter={(value) => (
                          <span style={{ color: "#D1D5DB" }}>{value}</span>
                        )}
                      />
                      <defs>
                        <linearGradient
                          id="barTotal"
                          x1="0"
                          y1="0"
                          x2="0"
                          y2="1"
                        >
                          <stop
                            offset="0%"
                            stopColor="#818cf8"
                            stopOpacity={0.7}
                          />
                          <stop
                            offset="100%"
                            stopColor="#818cf8"
                            stopOpacity={0.2}
                          />
                        </linearGradient>
                      </defs>
                      <Bar
                        yAxisId="left"
                        dataKey="count"
                        name="Total"
                        fill="url(#barTotal)"
                        radius={[3, 3, 0, 0]}
                        barSize={8}
                        animationDuration={CHART_ANIMATION_DURATION}
                      />
                      <Line
                        yAxisId="right"
                        type="monotone"
                        dataKey="uniqueCount"
                        name="Unique"
                        stroke="#a78bfa"
                        strokeWidth={2}
                        dot={false}
                        activeDot={{ r: 5, strokeWidth: 1, stroke: "#fff" }}
                        animationDuration={CHART_ANIMATION_DURATION}
                        animationBegin={200}
                      />
                    </ComposedChart>
                  )}
                </ResponsiveContainer>
              </div>
            </div>
          </motion.div>

          {/* Geography */}
          <motion.div
            className="relative overflow-hidden bg-gradient-to-br from-gray-900/70 to-indigo-950/70 rounded-xl shadow-lg border border-indigo-500/20 backdrop-blur-md p-4 sm:p-5"
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ duration: 0.5, delay: 0.2 }}
          >
            <div className="absolute bottom-0 right-0 w-32 h-32 bg-blue-600/10 rounded-full filter blur-3xl opacity-50"></div>
            <div className="relative z-10">
              <div className="flex justify-between items-center mb-4">
                <div className="flex items-center space-x-2">
                  <Globe size={16} sm={18} className="text-blue-400" />
                  <h3 className="text-base sm:text-lg font-medium text-white">
                    Top Countries
                  </h3>
                </div>
                <div className="text-xs bg-blue-900/30 text-blue-300 rounded-full px-2 py-0.5 border border-blue-700/30">
                  {geographyData.length} countries
                </div>
              </div>
              <div className="h-72 flex flex-col justify-between">
                {" "}
                {/* Ensure content fits */}
                {geographyData.length > 0 ? (
                  <>
                    <div className="flex-grow h-40">
                      {" "}
                      {/* Pie chart area */}
                      <ResponsiveContainer width="100%" height="100%">
                        <PieChart>
                          <Pie
                            data={geographyData}
                            dataKey="count"
                            nameKey="country"
                            cx="50%"
                            cy="50%"
                            innerRadius={35}
                            outerRadius={65}
                            paddingAngle={3}
                            animationDuration={CHART_ANIMATION_DURATION}
                            animationBegin={100}
                          >
                            {geographyData.map((entry, index) => (
                              <Cell
                                key={`cell-geo-${index}`}
                                fill={COLORS[index % COLORS.length]}
                                stroke="#1f2937"
                                strokeWidth={1}
                              />
                            ))}
                          </Pie>
                          <Tooltip content={<SimplePieTooltip />} />
                        </PieChart>
                      </ResponsiveContainer>
                    </div>
                    <div className="mt-3 space-y-1.5 max-h-28 overflow-y-auto pr-1 hide-scrollbar">
                      {" "}
                      {/* List area */}
                      {geographyData.slice(0, 5).map(
                        (
                          entry,
                          index // Show top 5 in list
                        ) => (
                          <motion.div
                            key={`geo-list-${index}`}
                            className="flex items-center justify-between px-2 py-1 rounded-md hover:bg-indigo-900/20 transition-colors text-xs"
                            initial={{ opacity: 0, x: -10 }}
                            animate={{ opacity: 1, x: 0 }}
                            transition={{ delay: index * 0.05 }}
                          >
                            <div className="flex items-center">
                              <div
                                className="w-2 h-2 rounded-full mr-2"
                                style={{
                                  backgroundColor:
                                    COLORS[index % COLORS.length],
                                }}
                              ></div>
                              <span
                                className="text-gray-300 truncate max-w-[100px]"
                                title={entry.country}
                              >
                                {entry.country}
                              </span>
                            </div>
                            <div className="flex items-center gap-2">
                              <span className="text-gray-400 font-medium">
                                {entry.count.toLocaleString()}
                              </span>
                              <span className="px-1.5 py-0.5 rounded bg-indigo-900/50 text-indigo-300 border border-indigo-700/30 text-[10px]">
                                {(
                                  (entry.count / stats.totals.totalViews) *
                                  100
                                ).toFixed(0)}
                                %
                              </span>
                            </div>
                          </motion.div>
                        )
                      )}
                    </div>
                  </>
                ) : (
                  <div className="text-center text-gray-400 h-full flex flex-col items-center justify-center">
                    <Globe size={40} className="mx-auto mb-3 text-gray-500" />
                    <p>No geographical data</p>
                  </div>
                )}
              </div>
            </div>
          </motion.div>
        </div>

        {/* Devices & Sources */}
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-4 mb-6">
          {/* Devices */}
          <motion.div
            className="relative overflow-hidden bg-gradient-to-br from-gray-900/70 to-indigo-950/70 rounded-xl shadow-lg border border-indigo-500/20 backdrop-blur-md p-4 sm:p-5"
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ duration: 0.5, delay: 0.3 }}
          >
            <div className="absolute top-0 left-0 w-32 h-32 bg-purple-600/10 rounded-full filter blur-3xl opacity-50"></div>
            <div className="relative z-10">
              <div className="flex justify-between items-center mb-4">
                <div className="flex items-center space-x-2">
                  <Laptop size={16} sm={18} className="text-purple-400" />
                  <h3 className="text-base sm:text-lg font-medium text-white">
                    Devices
                  </h3>
                </div>
                <div className="text-xs bg-purple-900/30 text-purple-300 rounded-full px-2 py-0.5 border border-purple-700/30">
                  {deviceData.length} types
                </div>
              </div>
              <div className="h-72 flex flex-col justify-between">
                {deviceData.length > 0 ? (
                  <>
                    <div className="flex-grow h-40">
                      <ResponsiveContainer width="100%" height="100%">
                        <PieChart>
                          <Pie
                            data={deviceData}
                            dataKey="count"
                            nameKey="device"
                            cx="50%"
                            cy="50%"
                            innerRadius={35}
                            outerRadius={65}
                            paddingAngle={3}
                            animationDuration={CHART_ANIMATION_DURATION}
                            animationBegin={200}
                          >
                            {deviceData.map((entry, index) => (
                              <Cell
                                key={`cell-dev-${index}`}
                                fill={COLORS[index % COLORS.length]}
                                stroke="#1f2937"
                                strokeWidth={1}
                              />
                            ))}
                          </Pie>
                          <Tooltip content={<SimplePieTooltip />} />
                        </PieChart>
                      </ResponsiveContainer>
                    </div>
                    <div className="mt-3 space-y-1.5 max-h-28 overflow-y-auto pr-1 hide-scrollbar">
                      {deviceData.map((entry, index) => (
                        <motion.div
                          key={`dev-list-${index}`}
                          className="flex items-center justify-between px-2 py-1 rounded-md hover:bg-indigo-900/20 transition-colors text-xs"
                          initial={{ opacity: 0, x: -10 }}
                          animate={{ opacity: 1, x: 0 }}
                          transition={{ delay: index * 0.05 }}
                        >
                          <div className="flex items-center">
                            <div
                              className="w-2 h-2 rounded-full mr-2"
                              style={{
                                backgroundColor: COLORS[index % COLORS.length],
                              }}
                            ></div>
                            <span className="text-gray-300 capitalize">
                              {entry.device}
                            </span>
                          </div>
                          <div className="flex items-center gap-2">
                            <span className="text-gray-400 font-medium">
                              {entry.count.toLocaleString()}
                            </span>
                            <span className="px-1.5 py-0.5 rounded bg-indigo-900/50 text-indigo-300 border border-indigo-700/30 text-[10px]">
                              {(
                                (entry.count / stats.totals.totalViews) *
                                100
                              ).toFixed(0)}
                              %
                            </span>
                          </div>
                        </motion.div>
                      ))}
                    </div>
                  </>
                ) : (
                  <div className="text-center text-gray-400 h-full flex flex-col items-center justify-center">
                    <Laptop size={40} className="mx-auto mb-3 text-gray-500" />
                    <p>No device data</p>
                  </div>
                )}
              </div>
            </div>
          </motion.div>

          {/* Sources (Referrers) */}
          <motion.div
            className="relative overflow-hidden bg-gradient-to-br from-gray-900/70 to-indigo-950/70 rounded-xl shadow-lg border border-indigo-500/20 backdrop-blur-md p-4 sm:p-5"
            initial={{ opacity: 0, y: 20 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ duration: 0.5, delay: 0.4 }}
          >
            <div className="absolute bottom-10 right-10 w-32 h-32 bg-cyan-600/10 rounded-full filter blur-3xl opacity-50"></div>
            <div className="relative z-10">
              <div className="flex justify-between items-center mb-4">
                <div className="flex items-center space-x-2">
                  <ExternalLink size={16} sm={18} className="text-cyan-400" />
                  <h3 className="text-base sm:text-lg font-medium text-white">
                    Top Sources
                  </h3>
                </div>
                <div className="text-xs bg-cyan-900/30 text-cyan-300 rounded-full px-2 py-0.5 border border-cyan-700/30">
                  {sourceData.length} sources
                </div>
              </div>
              <div className="h-72 flex flex-col justify-between">
                {sourceData.length > 0 ? (
                  <>
                    <div className="flex-grow h-40">
                      <ResponsiveContainer width="100%" height="100%">
                        <PieChart>
                          <Pie
                            data={sourceData}
                            dataKey="count"
                            nameKey="source"
                            cx="50%"
                            cy="50%"
                            innerRadius={35}
                            outerRadius={65}
                            paddingAngle={3}
                            animationDuration={CHART_ANIMATION_DURATION}
                            animationBegin={300}
                          >
                            {sourceData.map((entry, index) => (
                              <Cell
                                key={`cell-src-${index}`}
                                fill={COLORS[index % COLORS.length]}
                                stroke="#1f2937"
                                strokeWidth={1}
                              />
                            ))}
                          </Pie>
                          <Tooltip content={<SimplePieTooltip />} />
                        </PieChart>
                      </ResponsiveContainer>
                    </div>
                    <div className="mt-3 space-y-1.5 max-h-28 overflow-y-auto pr-1 hide-scrollbar">
                      {sourceData.slice(0, 5).map((entry, index) => (
                        <motion.div
                          key={`src-list-${index}`}
                          className="flex items-center justify-between px-2 py-1 rounded-md hover:bg-indigo-900/20 transition-colors text-xs"
                          initial={{ opacity: 0, x: -10 }}
                          animate={{ opacity: 1, x: 0 }}
                          transition={{ delay: index * 0.05 }}
                        >
                          <div className="flex items-center">
                            <div
                              className="w-2 h-2 rounded-full mr-2"
                              style={{
                                backgroundColor: COLORS[index % COLORS.length],
                              }}
                            ></div>
                            <span
                              className="text-gray-300 truncate max-w-[100px]"
                              title={entry.source || "Direct"}
                            >
                              {entry.source || "Direct"}
                            </span>
                          </div>
                          <div className="flex items-center gap-2">
                            <span className="text-gray-400 font-medium">
                              {entry.count.toLocaleString()}
                            </span>
                            <span className="px-1.5 py-0.5 rounded bg-indigo-900/50 text-indigo-300 border border-indigo-700/30 text-[10px]">
                              {(
                                (entry.count / stats.totals.totalViews) *
                                100
                              ).toFixed(0)}
                              %
                            </span>
                          </div>
                        </motion.div>
                      ))}
                    </div>
                  </>
                ) : (
                  <div className="text-center text-gray-400 h-full flex flex-col items-center justify-center">
                    <ExternalLink
                      size={40}
                      className="mx-auto mb-3 text-gray-500"
                    />
                    <p>No referrer data</p>
                  </div>
                )}
              </div>
            </div>
          </motion.div>
        </div>
      </div>
    </div>
  );
};

export default ViewsAnalyticsDashboard;
